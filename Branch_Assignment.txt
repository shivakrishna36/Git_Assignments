1. git init
2. git add Filename.txt  (for adding one file)
3. git add .  (for adding all files)
4. git commit -m "Adds new features"
5. git commit -am "Adds features"  (This combines both add and commit)
5. git status  (it shows untracked files)
6. git help   (it shows commands)
8. git log  (shows all the commits)
9. git --grep  (shows all the commit messages)
10. ls -la    (shows all hidden files)
11. git log -n  (shows no of commits)
12. git log --author="name"  (shows commits according to author name)
13.git log --since=yyyy-mm-dd  (shows commits from since to till date)
14.git log --until=yyyy-mm-dd  (shows commits until that date)
15.cd .git  (changes directory)
16.cat HEAD  (shows content in HEAD file)
17. git branch  (shows all branches)
18. git config --global user.email="github@gmail.com"   (Set the mail)
19.git remote add origin https://.....project.git  (Alias name for Remote repository)
20.git push origin master (push file to Remote repository)
